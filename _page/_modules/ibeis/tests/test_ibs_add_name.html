    # add a duplicate

    try:
        nid_list = ibs.add_names(name_list)

        assert name_list_test == name_list, 'sanity check'

        assert len(list(set(nid_list))) == len(list(set(name_list))), 'num unique ids / names should be the same'
    except AssertionError as ex:
        print('\n\nTEST ERROR')

        raise
    return locals()

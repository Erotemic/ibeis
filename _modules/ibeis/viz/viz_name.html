    import ibeis
    ibs = ibeis.opendb(defaultdb='testdb1')
    nid = ut.get_argval('--nid', type_=int, default=None)
    tags = ut.get_argval('--tags', type_=list, default=None)

    in_image = not ut.get_argflag('--no-inimage')
    return ibs, aid_list, in_image
